HMAC Signature Verification
Signing Method
# Every outgoing POST includes:
X-Webhook-Signature: sha256={HMAC_HEX}

# Signed over the raw JSON body string using HMAC-SHA256
# Signing key: defined in WEBHOOK_SIGNING_KEY env var
# Default key (dev only): nestack-webhook-secret-key-2024
Verify in Node.js
const crypto = require('crypto'); function verifyWebhook(rawBody, signature) { const key = process.env.WEBHOOK_SIGNING_KEY; // Strip 'sha256=' prefix const hex = signature.replace('sha256=', ''); const expected = crypto .createHmac('sha256', key) .update(rawBody) // rawBody = Buffer or string .digest('hex'); return crypto.timingSafeEqual( Buffer.from(hex, 'hex'), Buffer.from(expected, 'hex') ); } // Express usage: app.post('/webhook', express.raw({ type: '*/*' }), (req, res) => { const sig = req.headers['x-webhook-signature']; if (!verifyWebhook(req.body, sig)) { return res.status(401).json({ error: 'Invalid signature' }); } // Process event... res.sendStatus(200); });
Verify in Python
import hmac, hashlib def verify_webhook(raw_body: bytes, signature: str, key: str) -> bool: hex_sig = signature.removeprefix("sha256=") expected = hmac.new( key.encode(), raw_body, hashlib.sha256 ).hexdigest() return hmac.compare_digest(hex_sig, expected)
Retry Schedule
# Attempt order for each event:
Attempt 1 — immediate on ingestion
Attempt 2 — +30 seconds after failure
Attempt 3 — +5 minutes after failure
Attempt 4 — +30 minutes after failure
# After 4 total attempts → status: dead
# Poll interval: every 5 seconds

# Server restart: in-flight events (next_attempt_at = sentinel)
# are recovered on startup and scheduled for +30s retry
